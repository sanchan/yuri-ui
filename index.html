Buttons

For buttons you can define the contextual type setting the type option. The valid type values are "default", "primary", "success", "info", "warning", "danger".

Also you can set an icon setting the icon option. You can use any prefix used by Glyphicon icons set. For example, if you want to use the glyphicon-star icon, the you just need to set icon: "star":

<button id="button-primary"></button>
Yuri.ui({
  view: "button",
  el: "#button-primary",
  type: "primary",
  text: "Primary",
  icon: "star"
});
<button id="button-primary" class="btn btn-primary"><span class="text">Primary  </span><span class="glyphicon glyphicon-star"></span></button>

Alignment

By default the dropdown menu is left aligned, but you can align it to the right adding the option menuAlign: "right":

Yuri.ui({
  view: "dropdown",
  ...
  menuAlign: "right"
  ...
});

Headers

You can define a menu item as a menu header adding the option type: "header" to the item:

menu: [
  {type: "header", text: "Menu Header 1"},
  ...
]

Disabled menu items

You can disable a menu item adding the option disabled: true to the item:

menu: [
  {text: "Menu Header 1", disabled: true},
  ...
]

    Progress Bar
    
      
    Basic example
    
      
    
        
    
      
    
      
    
        
    
          
    <div id="progress-bar"></div>
    
        
    

        
    
          
    Yuri.ui({
  view: "progress-bar",
  el: "#progress-bar",
  value: 80
});
    
        
    

        
    
          
    <div id="progress-bar" class="progress">
  <div class="progress-bar" role="progressbar" aria-valuenow="80" aria-valuemin="0" aria-valuemax="100" style="transition: width 200ms ease-in-out; -webkit-transition: width 200ms ease-in-out; width: 80%;"><span class="sr-only">80%</span></div>
</div>
    
        
    
      
    

      
    With label
    
      
    You can enable the progress value label adding the option withLabel: true. Also you can define the format of the label string defining the label option:
    
      
    
        
    
      
    
      
    
        
    
          
    <div id="progress-bar-with-label"></div>
    
        
    

        
    
          
    Yuri.ui({
  view: "progress-bar",
  el: "#progress-bar-with-label",
  value: 80,
  withLabel: true,
  label: function (value) {
    return value + "% Complete"
  }
});
    
        
    

        
    
          
    <div id="progress-bar-with-label" class="progress">
  <div class="progress-bar" role="progressbar" aria-valuenow="80" aria-valuemin="0" aria-valuemax="100" style="transition: width 200ms ease-in-out; -webkit-transition: width 200ms ease-in-out; width: 80%;">80% Complete</div>
</div>
    
        
    
      
    

      
    Contextual alternatives
    
      
    You can define contextual alternatives to the progress bars defining the type option. The valid values are "success", "info", "warning", "danger":
    
      
    
        
    
        
    
        
    
        
    
      
    
      
    
        
    
          
    <div id="progress-bar-success"></div>
    
        
    

        
    
          
    Yuri.ui({
  view: "progress-bar",
  el: "#progress-bar-success",
  value: 40,
  type: "success"
});
    
        
    

        
    
          
    <div id="progress-bar-success" class="progress">
  <div class="progress-bar progress-bar-success" role="progressbar" aria-valuenow="40" aria-valuemin="0" aria-valuemax="100" style="transition: width 200ms ease-in-out; -webkit-transition: width 200ms ease-in-out; width: 40%;"><span class="sr-only">40%</span></div>
</div>
    
        
    
      
    

      
    Striped
    
      
    You can render the striped version of the bars setting the option striped: true:
    
      
    
        
    
        
    
        
    
        
    
      
    
      
    
        
    
          
    <div id="progress-bar-success-striped"></div>
    
        
    

        
    
          
    Yuri.ui({
  view: "progress-bar",
  el: "#progress-bar-success-striped",
  value: 40,
  type: "success",
  striped: true
});
    
        
    

        
    
          
    <div id="progress-bar-success-striped" class="progress">
  <div class="progress-bar progress-bar-success progress-bar-striped" role="progressbar" aria-valuenow="40" aria-valuemin="0" aria-valuemax="100" style="transition: width 200ms ease-in-out; -webkit-transition: width 200ms ease-in-out; width: 40%;"><span class="sr-only">40%</span></div>
</div>
    
        
    
      
    

      
    Active
    
      
    You can set the progress bar as active setting the options active: true and striped: true. This creates an animated striped bar:
    
      
    
        
    
      
    
      
    
        
    
          
    <div id="progress-bar-active"></div>
    
        
    

        
    
          
    Yuri.ui({
  view: "progress-bar",
  el: "#progress-bar-active",
  value: 40,
  type: "success",
  active: true,
  striped: true
});
    
        
    

        
    
          
    <div id="progress-bar-active" class="progress">
  <div class="progress-bar progress-bar-success active progress-bar-striped" role="progressbar" aria-valuenow="40" aria-valuemin="0" aria-valuemax="100" style="transition: width 200ms ease-in-out; -webkit-transition: width 200ms ease-in-out; width: 40%;"><span class="sr-only">40%</span></div>
</div>
    
        
    
      
    

      
    Stacked
    
      
    You can stack multiple bars in the same progress bar defining the value option as an array of bars:
    
      
    
        
    
      
    
      
    
        
    
          
    <div id="progress-bar-stacked"></div>
    
        
    

        
    
          
    Yuri.ui({
  view: "progress-bar",
  el: "#progress-bar-stacked",
  value: [
    {value: 35, type: "success"},
    {value: 20, type: "warning", striped: true},
    {value: 10, type: "danger"}
  ],
  type: "success",
  active: true,
  striped: true
});
    
        
    

        
    
          
    <div id="progress-bar-stacked" class="progress">
  <div class="progress-bar progress-bar-success" role="progressbar" aria-valuenow="35" aria-valuemin="0" aria-valuemax="100" style="width: 35%;"><span class="sr-only">35%</span></div>
  <div class="progress-bar progress-bar-warning progress-bar-striped" role="progressbar" aria-valuenow="20" aria-valuemin="0" aria-valuemax="100" style="width: 20%;"><span class="sr-only">20%</span></div>
  <div class="progress-bar progress-bar-danger" role="progressbar" aria-valuenow="10" aria-valuemin="0" aria-valuemax="100" style="width: 10%;"><span class="sr-only">10%</span></div>
</div>

    
        
    
      
    

      
    Animated
    
      
    You can change the animation duration of the progress bar when you update the value of the progress setting the animationDuration option. By default, every progress bar is animated, you can disable the animation setting the option animate: false. Note that this is not part of the Bootstrap's behavior:
    
      
    
        
    
      
    
      
    
        
    
          
    <div id="progress-bar-animated"></div>
    
        
    

        
    
          
    Yuri.ui({
  view: "progress-bar",
  el: "#progress-bar-animated",
  striped: true,
  active: true,
  animationDuration: 1000,
  withLabel: true,
  value: 80
});
    
        
    

        
    
          
    <div id="progress-bar-animated" class="progress">
  <div class="progress-bar active progress-bar-striped" role="progressbar" aria-valuenow="27" aria-valuemin="0" aria-valuemax="100" style="transition: width 1000ms ease-in-out; -webkit-transition: width 1000ms ease-in-out; width: 27%;">27%</div>
</div>
